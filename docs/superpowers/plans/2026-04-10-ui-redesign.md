# UI Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace Carelog's stacked-column layout with a clean icon-sidebar navigation, shadcn/ui component primitives, a custom Tailwind v4 design token layer, and full responsive support.

**Architecture:** A new `SidebarContext` controls which panel destination is active. A shared `SidebarNav` component renders in both a fixed desktop rail (`SidebarRail`) and a mobile Sheet drawer (`SidebarSheet`). `JournalClient` switches rendered panels based on the active destination. All panels are reskinned using shadcn `Card`/`Button`/`Badge` primitives with no changes to internal logic.

**Tech Stack:** Next.js 16, React 19, Tailwind CSS v4, shadcn/ui, Vitest + @testing-library/react, Playwright (e2e)

**Spec:** `docs/superpowers/specs/2026-04-10-ui-redesign-design.md`

```tsx
'use client'

import { createContext, useState } from 'react'

export type Destination =
  | 'journal'
  | 'medications'
  | 'team'
  | 'shifts'
  | 'documents'
  | 'more'

type SidebarContextType = {
  activeDestination: Destination
  setActiveDestination: (dest: Destination) => void
}

export const SidebarContext = createContext<SidebarContextType>({
  activeDestination: 'journal',
  setActiveDestination: () => {},
})

export function SidebarProvider({ children }: { children: React.ReactNode }) {
  const [activeDestination, setActiveDestination] = useState<Destination>('journal')
  return (
    <SidebarContext.Provider value={{ activeDestination, setActiveDestination }}>
      {children}
    </SidebarContext.Provider>
  )
}
```

```tsx
'use client'

import { useContext } from 'react'
import { SidebarContext } from './SidebarContext'
import type { Destination } from './SidebarContext'

const NAV_ITEMS: { dest: Destination; label: string; icon: string }[] = [
  { dest: 'journal',     label: 'Journal',     icon: '📋' },
  { dest: 'medications', label: 'Medications', icon: '💊' },
  { dest: 'team',        label: 'Team',        icon: '👥' },
  { dest: 'shifts',      label: 'Shifts',      icon: '📅' },
  { dest: 'documents',   label: 'Documents',   icon: '📁' },
  { dest: 'more',        label: 'More',        icon: '⋯' },
]

type Props = {
  /** When true, renders icon + label (mobile drawer). When false, icon only (desktop rail). */
  showLabels?: boolean
  onNavigate?: () => void
}

export function SidebarNav({ showLabels = false, onNavigate }: Props) {
  const { activeDestination, setActiveDestination } = useContext(SidebarContext)

  function handleClick(dest: Destination) {
    setActiveDestination(dest)
    onNavigate?.()
  }

  return (
    <nav className="flex flex-col gap-1">
      {NAV_ITEMS.map(({ dest, label, icon }) => {
        const isActive = activeDestination === dest
        return (
          <button
            key={dest}
            aria-current={isActive ? 'page' : undefined}
            onClick={() => handleClick(dest)}
            className={[
              'flex items-center gap-3 rounded-lg transition-colors',
              showLabels ? 'px-3 py-2 w-full text-left' : 'w-10 h-10 justify-center mx-auto',
              isActive
                ? 'bg-[var(--color-sidebar-active)] border border-brand/40 text-white'
                : 'text-slate-400 hover:text-slate-200 hover:bg-[var(--color-sidebar-item)]',
            ].join(' ')}
          >
            <span className="text-base leading-none">{icon}</span>
            {showLabels && (
              <span className="text-sm font-medium">{label}</span>
            )}
          </button>
        )
      })}
    </nav>
  )
}
```

```ts
import { test, expect } from '@playwright/test'

// Smoke test: verifies the redesigned layout renders without errors
// after a logged-in user navigates to their journal.
// This test requires a seeded test user — see e2e/CLAUDE.md for setup.

test.describe('UI layout smoke', () => {
  test('sidebar rail is present on desktop', async ({ page }) => {
    // Authenticate — adjust selector to match your auth helper
    await page.goto('/signin')
    await page.fill('[name="email"]', process.env.E2E_USER_EMAIL ?? '')
    await page.fill('[name="password"]', process.env.E2E_USER_PASSWORD ?? '')
    await page.click('[type="submit"]')
    await page.waitForURL(/\/journal\//)

    // Sidebar rail should be present
    await expect(page.getByTestId('sidebar-rail')).toBeVisible()

    // Top bar should be present
    await expect(page.getByTestId('top-bar')).toBeVisible()

    // No console errors
    const errors: string[] = []
    page.on('console', msg => {
      if (msg.type() === 'error') errors.push(msg.text())
    })
    await page.waitForTimeout(1000)
    expect(errors.filter(e => !e.includes('favicon'))).toHaveLength(0)
  })

  test('hamburger menu opens sidebar on mobile', async ({ page }) => {
    await page.setViewportSize({ width: 390, height: 844 })
    await page.goto('/signin')
    await page.fill('[name="email"]', process.env.E2E_USER_EMAIL ?? '')
    await page.fill('[name="password"]', process.env.E2E_USER_PASSWORD ?? '')
    await page.click('[type="submit"]')
    await page.waitForURL(/\/journal\//)

    // Sidebar rail should be hidden on mobile
    await expect(page.getByTestId('sidebar-rail')).not.toBeVisible()

    // Hamburger button should be visible
    const hamburger = page.getByRole('button', { name: /menu/i })
    await expect(hamburger).toBeVisible()

    // Opening the sheet shows nav items
    await hamburger.click()
    await expect(page.getByText('Carelog')).toBeVisible()
  })
})
```
